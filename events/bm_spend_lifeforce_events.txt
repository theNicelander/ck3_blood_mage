    type = character_event
    title = channel_lifeforce_decision
    desc = channel_lifeforce_decision_tooltip_desc
    theme = witchcraft

        name = "I want to learn faster"

            modifier = exp_lifeforce_modifier

    }

    option = {

        ai_chance = { base = 10 }

        trigger_event = { id = bm_channel_lifeforce.0002 }
    }

    # Portraits
    left_portrait ={
        character = root
        animation = ecstasy
    }
}
