	type = character_event
	title = bm_lifedrain.001.title
	desc = bm_lifedrain.001.desc

	option = {

	option = {
		name = bm_lifedrain.001.b
		trigger = {

		}
	}

	option = {
		name = bm_lifedrain.001.c

			scope:recipient = {

			}
		}
		scope:actor = {
			# Carry out logic
			physique_drain_trait_effects = yes
		}

			VICTIM = scope:recipient

		}
