intellect_drain_trait_effects = {
    if = {
        limit = { NOT = { has_trait = intellect_good } }
        add_trait = intellect_good_1
    }
    else = {
        change_trait_rank = {
            trait = intellect_good
            rank = 1
            max = 5
        }
    }
}
