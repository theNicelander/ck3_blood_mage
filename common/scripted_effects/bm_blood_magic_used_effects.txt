temporary_negative_lifeforce_effect = {
    add_character_modifier = {
        modifier = lifeforce_modifier_negative
        days = 365
    }
}

blood_magic_used = {
    remove_character_modifier = lifeforce_modifier
    add_character_modifier = {
        modifier = lifeforce_modifier_negative
        days = 365
    }
    # Grant exp when using magic
    if = {
        limit = {
            has_trait = lifestyle_blood_mage_0
        }
        add_trait_xp = {
            trait = lifestyle_blood_mage_0
            value = 1
        }
    }
    if = {
        limit = {
            has_trait = lifestyle_blood_mage_1
        }
        add_trait_xp = {
            trait = lifestyle_blood_mage_1
            value = 1
        }
    }
    if = {
        limit = {
            has_trait = lifestyle_blood_mage_2
        }
        add_trait_xp = {
            trait = lifestyle_blood_mage_2
            value = 1
        }
    }
}
