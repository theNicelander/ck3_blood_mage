enhance_blood_mage_coven_decision_1 = {
    picture = { reference = "gfx/interface/illustrations/decisions/decision_major_religion.dds" }
    ai_check_interval = 36
    desc = enhance_blood_mage_coven_decision_desc
    decision_group_type = bm_decision_group

    is_shown = {
        has_trait = lifestyle_blood_mage_group
        exists = house
        is_house_head = yes

        NOT = {
            exists = global_var:enhance_bm_1
        }
    }

    is_valid = {
        is_house_head = yes
        has_character_modifier = lifeforce_modifier

        house = {
            custom_tooltip = {
                text = found_bm_coven_member_count_tt
                any_house_member = {
                    count >= found_bm_coven_member_count
                    has_trait = lifestyle_blood_mage_group
                }
            }
            any_house_member = {
                count >= found_bm_coven_member_count
                has_trait = lifestyle_blood_mage_group
            }
        }
    }

    effect = {
        blood_magic_used_set_exp = { EXP = 1 }
        enhance_blood_mage_coven_decision_effects = yes
        set_variable = { name = enhance_bm_1 }
    }

    cost = {
        piety = 100
    }

    ai_will_do = { base = 100 }
}
