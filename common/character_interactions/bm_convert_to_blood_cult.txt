convert_to_blood_cult = {
    common_interaction = yes
    interface_priority = 300
    category = interaction_category_friendly
    desc = "Convert to Blood Cult"

    popup_on_receive = yes
    pause_on_receive = yes

    auto_accept = yes
    is_highlighted = { always = yes }

    cost = { piety = 35 }

    is_shown = {
        scope:actor = {
            has_trait = lifestyle_blood_mage
            has_faith = faith:cult_of_blood
        }
        scope:recipient = {
            NOT = { has_faith = faith:cult_of_blood }
            OR = {
                is_imprisoned_by = scope:actor
                is_courtier_of = scope:actor
            }
        }
    }

    is_valid_showing_failures_only = {
        scope:actor = {
            is_imprisoned = no
            has_character_modifier = lifeforce_modifier_minor
        }
    }

    on_accept = {
        scope:recipient = {
            send_interface_toast = {
                left_icon = scope:actor
                custom_tooltip = convert_to_blood_cult_event_interaction.recipient.notification.tt
            }
        }
        scope:actor = {
            blood_magic_used_minor = yes
            add_xp_benediction = yes
            duel = {
                skill = learning
                target = scope:recipient

                50 = {
                    desc = "I successfully indoctrinate them into the Blood Cult"
                    compare_modifier = {
                        value = scope:duel_value
                        multiplier = 3
                    }
                    scope:recipient = { set_character_faith = faith:cult_of_blood }
                    send_interface_toast  = {
                        title = "[recipient.GetShortUINameNoTooltip] was successfully converted"
                        left_icon = scope:actor
                        right_icon = scope:recipient
                    }
                }
                50 = {
                    desc = "I fail"
                    send_interface_toast  = {
                        title = "[recipient.GetShortUINameNoTooltip] was not converted"
                        left_icon = scope:actor
                        right_icon = scope:recipient
                    }
                }
            }
        }

    }
    ai_will_do = {
        base = -10
        modifier = {
            scope:recipient = { is_courtier_of = scope:actor }
            add = 30
        }
        modifier = {
            is_knight_of = scope:actor
            add = 30
        }
        modifier = {
            scope:recipient = { is_powerful_vassal_of = scope:actor }
            add = 30
        }
    }
}
