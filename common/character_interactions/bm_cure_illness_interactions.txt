heal_disease_minor = {
	interface_priority = 200
	category = interaction_category_friendly
	desc = heal_disease_minor_desc
	common_interaction = yes

	cost = { piety = heal_disease_minor_cost }
	is_shown = {
		scope:actor = { has_trait = blood_mage }
		scope:recipient = {
			OR = {				
				has_trait = wounded_1
				has_trait = ill
				has_trait = lovers_pox
				has_trait = gout_ridden
			}
		}
	}

	on_accept = {
		scope:actor = { magic_used = yes }
		scope:recipient = {
			send_interface_toast = { left_icon = scope:actor }
			remove_trait = wounded_1
			remove_trait = ill
			remove_trait = lovers_pox
			remove_trait = gout_ridden

			add_opinion = {
				target = scope:actor
				modifier = magic_opinion_positive
				opinion = 20
			}
		}
	}
	auto_accept = yes
}
