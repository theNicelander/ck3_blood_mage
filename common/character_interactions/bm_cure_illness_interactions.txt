heal_disease_minor = {
    interface_priority = 210
    category = interaction_category_friendly
    desc = heal_disease_minor_desc
    common_interaction = yes

    cost = { piety = heal_disease_minor_cost }
    is_shown = {
        scope:actor = {
            has_trait = lifestyle_blood_mage_group
        }
        scope:recipient = {
            OR = {
                has_trait = wounded_1
                has_trait = ill
                has_trait = lovers_pox
                has_trait = gout_ridden
            }
        }
    }

    is_valid_showing_failures_only = { scope:actor = { has_character_modifier = lifeforce_modifier } }

    on_accept = {
        scope:actor = { blood_magic_used = yes }
        scope:recipient = {
            remove_trait = wounded_1
            remove_trait = ill
            remove_trait = lovers_pox
            remove_trait = gout_ridden

            # temporary positive opinion
            add_opinion = {
                target = scope:actor
                modifier = magic_opinion_positive
                opinion = 20
            }
        }
    }
    auto_accept = yes
}
